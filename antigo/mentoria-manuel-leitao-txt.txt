    function formatNumber(n, prefix, suffix, format) {
      let display;
      if (format === 'M') {
        if (n >= 1000000) display = (n / 1000000).toFixed(1).replace('.0','') + 'M+';
        else if (n >= 100000) display = (n / 1000000).toFixed(2).replace(/\.?0+$/, '') + 'M';
        else display = (n / 1000).toFixed(0) + 'k';
      } else if (n >= 10000) {
        display = n.toLocaleString('pt-PT') + (suffix || '');
      } else {
        display = n + (suffix || '');
      }
      return (prefix || '') + display;

        if (tx >= rect.left && tx <= rect.right) setPos(tx);
      }, { passive: true });
      document.addEventListener('touchend', () => { dragging = false; });

      // Animate intro — slide from 100% to 50%
      let introPos = 95;
      reveal.style.setProperty('--pos', introPos + '%');
      function introAnim() {
        if (introPos <= 50) return;
        introPos = Math.max(50, introPos - 1.5);
        reveal.style.setProperty('--pos', introPos + '%');
        requestAnimationFrame(introAnim);
      }
      const obs = new IntersectionObserver(entries => {
        if (entries[0].isIntersecting) { obs.disconnect(); setTimeout(introAnim, 400); }
      }, { threshold: 0.4 });
      obs.observe(reveal);
    })();

      const obs = new IntersectionObserver((entries) => {
        entries.forEach(entry => {

            // Meta Pixel — ViewContent
            if (typeof fbq !== 'undefined') fbq('track', 'ViewContent');
            // GA4 — scroll_to_form
            if (typeof gtag !== 'undefined') gtag('event', 'scroll_to_form');

          }
        });

    // ─── INSPIRAÇÕES CAROUSEL ───
    (function() {
      const track = document.getElementById('inspoTrack');
      const prev  = document.getElementById('inspoPrev');
      const next  = document.getElementById('inspoNext');
      const dotsEl= document.getElementById('inspoDots');
      if (!track) return;

      const slides = track.querySelectorAll('.inspo-slide');
      const total  = slides.length;
      let current  = 0;
      let visCount = window.innerWidth <= 600 ? 1 : 2;

      function slideWidth() {
        const gap = 16;
        return (track.parentElement.offsetWidth / visCount) - gap;
      }

      function goTo(idx) {
        const max = total - visCount;
        current = Math.max(0, Math.min(idx, max));
        track.style.transform = `translateX(-${current * (slideWidth() + 16)}px)`;
        dotsEl.querySelectorAll('.inspo-dot').forEach((d, i) => {
          d.classList.toggle('inspo-dot--active', i === current);
        });
      }

      prev && prev.addEventListener('click', () => goTo(current - 1));
      next && next.addEventListener('click', () => goTo(current + 1));
      dotsEl && dotsEl.querySelectorAll('.inspo-dot').forEach((d, i) => {
        d.addEventListener('click', () => goTo(i));
      });

      window.addEventListener('resize', () => {
        visCount = window.innerWidth <= 600 ? 1 : 2;
        goTo(Math.min(current, total - visCount));
      });
    })();

          node = node.parentElement;
        }
      }

      setTimeout(fixParents, 300);
      setTimeout(fixParents, 1000);
      window.addEventListener('load', fixParents);
    })();
